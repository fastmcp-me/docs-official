```typescript TypeScript
import { 
  GalileoScorers, getDataset, 
  getPromptTemplate, runExperiment
} from "galileo";

// Get an existing dataset
const dataset = await getDataset({
    name: "countries"
});

// Get an existing prompt template
const promptTemplate = await getPromptTemplate({
    projectName: "my-project",
    name: "geography-prompt",
  });

// Run an experiment with the dataset and prompt
await runExperiment({
    name: "geography-experiment",
    dataset: dataset,
    promptTemplate: promptTemplate,
    metrics: [GalileoScorers.Correctness],
    projectName: "my-project",
  });
```
