import { runExperiment, GalileoScorers, getPromptTemplate, getDataset } from "galileo";

async function runPromptExperiment() {
  const template = await getPromptTemplate({
    projectName: "my-project",
    name: "storyteller-prompt",
  });

  const dataset = await getDataset(undefined, "storyteller-dataset");

  await runExperiment({
    name: "Test Experiment",
    dataset: dataset,
    promptTemplate: template,
    metrics: [GalileoScorers.InputToxicity, GalileoScorers.OutputToxicity],
    projectName: "my-project",
  });
}
